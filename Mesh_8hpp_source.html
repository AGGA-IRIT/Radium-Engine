417 template <typename CoreMeshType>
418 CoreMeshType createCoreMeshFromGeometryData( const Ra::Core::Asset::GeometryData* data ) {
419  CoreMeshType mesh;
420  typename CoreMeshType::PointAttribHandle::Container vertices;
421  typename CoreMeshType::NormalAttribHandle::Container normals;
422  typename CoreMeshType::IndexContainerType indices;
423 
424  vertices.reserve( data->getVerticesSize() );
425  std::copy(
426  data->getVertices().begin(), data->getVertices().end(), std::back_inserter( vertices ) );
427 
428  if ( data->hasNormals() )
429  {
430  normals.reserve( data->getVerticesSize() );
431  std::copy(
432  data->getNormals().begin(), data->getNormals().end(), std::back_inserter( normals ) );
433  }
434 
435  const auto& faces = data->getFaces();
436  indices.reserve( faces.size() );
437  std::copy( faces.begin(), faces.end(), std::back_inserter( indices ) );
438  mesh.setVertices( std::move( vertices ) );
439  mesh.setNormals( std::move( normals ) );
440 
441  // \todo remove when data will handle all the attributes in a coherent way.
442  if ( data->hasTangents() )
443  {
444  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_TANGENT ),
445  data->getTangents() );
446  }
447 
448  if ( data->hasBiTangents() )
449  {
450  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_BITANGENT ),
451  data->getBiTangents() );
452  }
453 
454  if ( data->hasTextureCoordinates() )
455  {
456  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_TEXCOORD ),
457  data->getTexCoords() );
458  }
459 
460  if ( data->hasColors() )
461  {
462  mesh.addAttrib( Data::Mesh::getAttribName( Data::Mesh::VERTEX_COLOR ), data->getColors() );
463  }
464 
465  // add custom attribs
466  // only attributs not handled before are handled by data->getAttribManager()
467  // but futur plan will handle also "usual" attibutes this way
468  mesh.vertexAttribs().copyAllAttributes( data->getAttribManager() );
469 
470  // also this one will be automatically done with the futur behavior
471  // mesh->addData( Data::Mesh::VERTEX_WEIGHTS, meshData.weights );
472 
473  mesh.setIndices( std::move( indices ) );
474 
475  return mesh;
476 }
