305 void Gui::TrackballCameraManipulator::handleCameraRotate( Scalar dx, Scalar dy ) {
306  Scalar dphi = dx * m_cameraSensitivity * m_quickCameraModifier;
307  Scalar y = -dy * m_cameraSensitivity * m_quickCameraModifier;
308 
309  Scalar phi = std::fmod( m_phi + dphi, 2_ra * Pi ); // Keep phi between 0 and 2pi
310  Scalar theta = std::fmod( m_theta + y + Pi, 2_ra * Pi ) - Pi; // Keep theta between -pi and pi
311 
312  Scalar dtheta = theta - m_theta;
313 
314  Core::Vector3 P =
315  m_target + m_distFromCenter * Core::Vector3( -std::cos( phi ) * std::sin( theta ),
316  std::cos( theta ),
317  -std::sin( phi ) * std::sin( theta ) );
318 
319  Core::Vector3 t( P - m_camera->getPosition() );
320 
321  // Translate the camera given this translation
322  Core::Transform T( Core::Transform::Identity() );
323  T.translation() = t;
324 
325  // Rotate the camera so that it points to the center
326  Core::Transform R1( Core::Transform::Identity() );
327  Core::Transform R2( Core::Transform::Identity() );
328 
329  Core::Vector3 U = Core::Vector3( 0_ra, 1_ra, 0_ra );
330  Core::Vector3 R = -m_camera->getRightVector().normalized();
331 
332  R1 = Core::AngleAxis( -dphi, U );
333  R2 = Core::AngleAxis( -dtheta, R );
334 
335  m_camera->applyTransform( T * R1 * R2 );
336  m_phi = phi;
337  m_theta = theta;
338 }

342  Scalar y = dy * m_cameraSensitivity * m_quickCameraModifier * m_distFromCenter * 0.1_ra;

382 void Gui::TrackballCameraManipulator::updatePhiTheta() {
383  using Core::Math::areApproxEqual;
384  const auto R = m_camera->getDirection().normalized();
385 
386  m_theta = std::acos( -R.y() );
387 
388  m_phi = ( areApproxEqual( R.z(), 0_ra ) && areApproxEqual( R.x(), 0_ra ) )
389  ? std::acos( m_camera->getRightVector().dot( Ra::Core::Vector3::UnitZ() ) ) + Pi
390  : std::atan2( R.z(), R.x() );
391 
392  // Keep phi between 0 and 2pi
393  // Keep theta between -pi and pi
394  if ( m_phi < 0_ra )
395  {
396  m_theta = -m_theta + Pi;
397  m_phi += 2_ra * Pi;
398  }
399 
400  CORE_ASSERT( std::isfinite( m_theta ) && std::isfinite( m_phi ), "Error in trackball camera" );
401 }
