79 void Camera::updateProjMatrix() {
80 
81  switch ( m_projType )
82  {
83  case ProjType::ORTHOGRAPHIC: {
84  const Scalar dx = m_zoomFactor * .5_ra;
85  const Scalar dy = dx / m_aspect;
86  // ------------
87  // Compute projection matrix as describe in the doc of gluPerspective()
88  const Scalar l = -dx; // left
89  const Scalar r = dx; // right
90  const Scalar t = dy; // top
91  const Scalar b = -dy; // bottom
92 
93  Core::Vector3 tr( -( r + l ) / ( r - l ),
94  -( t + b ) / ( t - b ),
95  -( ( m_zFar + m_zNear ) / ( m_zFar - m_zNear ) ) );
96 
97  m_projMatrix.setIdentity();
98 
99  m_projMatrix.coeffRef( 0, 0 ) = 2_ra / ( r - l );
100  m_projMatrix.coeffRef( 1, 1 ) = 2_ra / ( t - b );
101  m_projMatrix.coeffRef( 2, 2 ) = -2_ra / ( m_zFar - m_zNear );
102  m_projMatrix.block<3, 1>( 0, 3 ) = tr;
103  }
104  break;
105 
106  case ProjType::PERSPECTIVE: {
107  // Compute projection matrix as describe in the doc of gluPerspective()
108  const Scalar f = std::tan( ( PiDiv2 ) - ( m_fov * m_zoomFactor * .5_ra ) );
109  const Scalar diff = m_zNear - m_zFar;
110 
111  m_projMatrix.setZero();
112 
113  m_projMatrix.coeffRef( 0, 0 ) = f / m_aspect;
114  m_projMatrix.coeffRef( 1, 1 ) = f;
115  m_projMatrix.coeffRef( 2, 2 ) = ( m_zFar + m_zNear ) / diff;
116  m_projMatrix.coeffRef( 2, 3 ) = ( 2_ra * m_zFar * m_zNear ) / diff;
117  m_projMatrix.coeffRef( 3, 2 ) = -1_ra;
118  }
119  break;
120 
121  default:
122  break;
123  }

144 void Camera::fitZRange( const Core::Aabb& aabb ) {
145  const auto& minAabb = aabb.min();
146  const auto& maxAabb = aabb.max();
147  const auto& position = m_frame.translation();
148  Ra::Core::Vector3 direction = m_frame.linear() * Ra::Core::Vector3( 0_ra, 0_ra, -1_ra );
149 
150  m_zNear = m_zFar = direction.dot( minAabb - position );
151 
152  auto adaptRange = [position, direction, this]( Scalar x, Scalar y, Scalar z ) {
153  Ra::Core::Vector3 corner( x, y, z );
154  auto d = direction.dot( corner - position );
155  this->m_zNear = std::min( d, this->m_zNear );
156  this->m_zFar = std::max( d, this->m_zFar );
157  };
158 
159  adaptRange( minAabb[0], minAabb[1], maxAabb[2] );
160  adaptRange( minAabb[0], maxAabb[1], minAabb[2] );
161  adaptRange( minAabb[0], maxAabb[1], maxAabb[2] );
162  adaptRange( maxAabb[0], maxAabb[1], maxAabb[2] );
163  adaptRange( maxAabb[0], maxAabb[1], minAabb[2] );
164  adaptRange( maxAabb[0], minAabb[1], maxAabb[2] );
